### 通过链表实现栈
#### 问题描述
> 通过链表实现栈，包括push, pop, empty, size, top等函数

#### 解决方案
**算法代码如下：**
```cpp
class LinkStack {
private:
    Node *s;
    int count;
public:
    LinkStack() {
        s = new Node;
        s->next = nullptr;
        count = 0;
    }

    void push(int data) {
        Node *tmp = new Node;
        tmp->value = data;
        tmp->next = s->next;
        s->next = tmp;
        ++count;
    }

    void pop() {
        if(s->next == nullptr) {
            return;
        }
        Node* tmp = s->next;
        s->next = s->next->next;
        --count;
        delete tmp;
    }

    bool empty() const {
        return (s->next == nullptr);
    }

    int& top() const {
        if (s->next == nullptr)
            throw "No data";
        return s->next->value;
    }

    int size() const {
        return count;
    }
};
```

**测试用例如下：**
```cpp
TEST(StackFifo, Class_Stack) {
    LinkStack s;
    s.push(1);
    s.push(2);
    s.push(3);
    s.push(4);
    EXPECT_EQ(4, s.top());
    EXPECT_EQ(4, s.size());
    EXPECT_FALSE(s.empty());

    s.pop();
    EXPECT_EQ(3, s.top());
    EXPECT_EQ(3, s.size());
    EXPECT_FALSE(s.empty());
    s.pop();
    s.pop();
    s.pop();
    s.pop();
    EXPECT_TRUE(s.empty());
    EXPECT_EQ(0, s.size());
    ASSERT_ANY_THROW(s.top());

    s.push(1);
    s.push(2);
    EXPECT_EQ(2, s.top());
    EXPECT_EQ(2, s.size());
    EXPECT_FALSE(s.empty());
}
```
